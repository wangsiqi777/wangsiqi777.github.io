1. 线性表

1.1 顺序表

EG：将十进制的数转化为任意进制的数

#include<iostream>

using namespace std;
int main(){
    int num;
    int n;
    cout<<"请输入数据："<<endl;
    cin>>num;
    cout<<"请输入要转换的进制"<<endl;
    cin>>n;
    
    int a[100];
    int i=0;
    while(num!=0){
        //存储余数
        a[i++]=num%n;
        //更新num
        num/=n;
}
    for(int j=i;j>=0;j--){
        cout<<a[j];
    }
    cout<<endl;
}

1.2 链表

单链表

双向链表

循环链表

头节点只用于标记开始节点

开始节点是链表真正开始的第一个节点

创建头节点是为了使得操作同构：比如插入一个节点

创建新节点——新节点指针指向下一个节点——前一个节点指针改为指向新节点

插入节点时注意是先将新节点指向后一个节点


#include <iostream>

using namespace std;

struct Node{
    int data;
    Node *next;
};

//创建:两种方法：头插法和尾插法
//头插法：首先创建头节点和尾节点，每次在头节点之后插入新节点
//尾插法：首先创建头节点和尾节点，每次将尾节点变成新节点再创建尾节点
//传入头节点和创建长度，创建完链表后，再返回头节点
Node *creatList(Node *head,int n){
    //创建工作指针
    Node *p=new Node;
    //头指针指向尾指针
    head->next=p;
    for(int i=2;i<=n;i++){
        //创建下一个节点
        Node *q=new Node;
        p->next=q;
        p=q;
}
    //尾指针指向NULL
    p->next=NULL;
    return head;
}

void initList(Node *head){
    //首先p来到head后的第一个节点起始节点
    Node *p=head->next;
    int i=1;
    //最后p为NULL时结束
    for(;p!=NULL;p=p->next){
        cout<<"请输入"<<i++<<"个节点的值"<<endl;
        cin>>p->data;
    }
}

void show(Node *head){
    Node *p=head->next;
    for(;p!=NULL;p=p->next){
        cout<<"这个节点值为："<<p->data<<endl;
    }
}

void del(Node *head,int pos){
    Node *p=head;
    //找到需删除位置的前驱节点
    for(int i=1;i<pos;i++){
        p=p->next;
    }
    //删除节点
    Node *q=p->next;
    //前驱节点指向需删除节点的后一个节点
    p->next=q->next;
    delete q;
}

void add(Node *head,int x,int pos){
    Node *p=head;
    //找到需插入位置的前驱节点
    for(int i=1;i<pos;i++){
        p=p->next;
    }
    //插入节点
    Node *q=p->next;
    Node *t;
    t->data=x;
    t->next=q;
    p->next=t;
}

int main(){
    //创建
    Node *head=new Node;
    head=creatList(head,5);
    //初始化
    initList(head);
    //访问 按值访问 按位访问
    //修改
    //遍历
    show();
    //删除节点
    del(head,3);
    //插入节点
    add(head,2,2);
}

EG:单链表的逆置

Node *function(Node *head){
    //从开始节点起
    Node *p=head->next;
    //开始节点的前驱置为NULL
    Node *pre=NULL;
    while(p!=NULL){
        //保存原始后继节点
        //前驱节点变后继节点
        Node q=p->next;
        p->next=pre;
        //更新下一个pre是自己
        pre=p;
        p=q; 
    }
    head->next=pre;
    return head;
}

2. 栈和队列

EG：生成一个长度为100的数组，采用随机函数生成数值，依次进栈，然后依次出栈

#include <iostream>

#include <ctime>

using namespace std;

int main(){
    srand(unsigned(NULL));
    int a[100];
    int top=-1;
    while(top<99){
        a[++top]=rand()%100+1;
    }
    while(top!=-1){
        cout<<a[top--]<<endl;
    }
    return 0;
}

3. 二叉树

//递归建立二叉树
#include <iostream>

using namespace std;

struct Tree{
    Tree *lchild;
    Tree *rchild;
    char data;
};

//前序创建树
Tree *creatTree(Tree *t){
    char ch;
    cin>>ch;
    //设#表示该节点为空
    if(ch=='#'){
        t=NULL;
    }
    else{
        t->data=ch;
        t->lchild=creatTree(t->lchild);
        t->rchild=creatTree(t->rchild);
    }
    return t;
}

int main(){
    Tree *t=new Tree;
    t=creatTree(t);
}
